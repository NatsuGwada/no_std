12/// J'ai choisi d'utiliser 4 tailles de slabs différentes pour ne pas gaspiller la ram.
13/// tu veux alouer 20 octés bin voila un slab de 32 comme ça tu es tranquille
14/// Définition de la taille d'un slab (slab = bloc mémoires en français)
15const SLAB_SIZES: [usize; 4] = [16, 32, 64, 128]; // Différentes tailles de slabs
16
17/// Cette structure représente un bloc de mémoire libre.
18/// Quand il est libre, on peut utiliser son espace pour stocker un pointeur
19/// vers le prochain bloc libre, et ça forme ainsi une liste chaînée.
20/// Voila, Un bloc de mémoire dans notre allocateur
21#[repr(C)]
22struct SlabBlock {
23    next: *mut SlabBlock, // Pointeur vers le prochain bloc libre
24}

27/// Notre allocateur de type Slab
28pub struct SlabAllocator {
29    /// Utilisation d'UnsafeCell pour permettre une mutabilité intérieure
30    /// tout en respectant les règles de Rust concernant les références mutables
31    inner: UnsafeCell<SlabAllocatorInner>,
32}
33
34/// Structure interne contenant les données de l'allocateur
35struct SlabAllocatorInner {
36    /// La mémoire que notre allocateur va gérer, debut et fin (les limites)
37    /// usize = valeurs d'adresse mémoire, en gros un pointeur non signé
38    memory_start: usize,  
39    memory_end: usize,
40    
41    /// Liste\tableu de blocs libres pour chaque taille de slab, SlabBlock = Block mémoire & slabsize sa taille
42    free_lists: [*mut SlabBlock; SLAB_SIZES.len()],
43    
44    /// Nombre de slabs alloués pour chaque taille
45    allocated_counts: [usize; SLAB_SIZES.len()],
46}
47
48impl SlabAllocator {
49    /// Crée un nouvel allocateur
50    /// 
51    /// # Safety
52    /// 
53    /// La région de mémoire fournie doit être valide et non utilisée par d'autres parties du programme.
54    /// L'appelant doit garantir que la mémoire spécifiée par `start` et `size`
55    /// est valide, libre, et ne sera pas utilisée ailleurs dans le programme. 
56    /// une constante car elle va etre evaluer a la compilation pour les test et comme c'est du no_std donc il y a des initialisation a la compilation
57    pub const fn new(start: usize, size: usize) -> Self {
58        SlabAllocator {
59            // Inner = novuelle instance embalé sans unsafecell
60            inner: UnsafeCell::new(SlabAllocatorInner {
61                // Démarrage de l'attribution de taille mémoire
62                memory_start: start,
63                memory_end: start + size,
64                // Plage mémoire défini
65                free_lists: [ptr::null_mut(); SLAB_SIZES.len()],
66                allocated_counts: [0; SLAB_SIZES.len()],
67            }),
68        }
69    }
70    
71    /// Initialise les listes de blocs libres
72    /// 
73    /// /// # Safety
74    /// unsafe va modifier l'interieur de UnsafeCell
75    /// &self lis les données de la structure sans les modifiers, en gros de la lecture seule mais avec unsafeCell tu passes en lecture écriture donc tu peux modifier
76    pub unsafe fn init(&self) {
77        // self.inner donne un pointeur brut mutable sur &mut SlabAllocatorInner, d'ailleurs il est deviner tout seul par rust (en gros le mot est grisé dans mon vscode)
78        let inner = &mut *self.inner.get();
79        // on parcourt les index et les tailles des blocs (16, 32, 64, 128)
80        for (i, &_size) in SLAB_SIZES.iter().enumerate() {
81            // ça c'est pour vidé la liste de bloc libres
82            inner.free_lists[i] = ptr::null_mut();
83            // ça c'est pour remettre a 0 le compteur de bloc alloué pour la taille choisi
84            inner.allocated_counts[i] = 0;
85        }
86    }
87    
88    /// Trouve l'index de la taille de slab appropriée , en gros ça choisi quelle taile de bloc conviens a la demande d'allocation doonée 
89    ///
90    /// # Safety
91    /// on reviens a l'exemple d'en haut 20 octer tu veux? ok tien voila 32 octer de dispo pour toi
92    unsafe fn find_slab_index(&self, layout: &Layout) -> Option<usize> {
93        // let inner = &*self.inner.get();
94        // bon ça , ça calculle la taille minimal nécéssaire
95        let required_size = layout.size().max(mem::size_of::<SlabBlock>());
96        // La on parcours les 4 tailles possible, si compatibilité (sois les slab est assser grand ou pas), puis on retourn l'index si c'est bon sinon on renvoie None
97        // l'index c'est la position de la taille dans le tableau [16, 32, 64, 128], 16 = index 0, 32 = index 1, ect..
98        for (i, &size) in SLAB_SIZES.iter().enumerate() {
99            if size >= required_size && layout.align() <= size {
100                // renvoie de l'index ici
101                return Some(i);
102            }
103        }
104        
105        None // Aucune taille de slab appropriée, normalement il y a que une taille trop grosse par rapport a 128 qui ne passe pas
106    }
107    
108    /// Cette fonction est unsafe car elle manipule directement des pointeurs pour alouer plus de blocs libres pour un slab donnée
109    /// et accède à la mémoire sans vérification.
110    ///
111    /// # Safety
112    ///
113    /// Le code appelant doit garantir que:
114    /// - L'entrée `slab_index` est valide et dans les limites de `SLAB_SIZES`
115    /// - La région mémoire spécifiée par `memory_start` et `memory_end` est valide
116    /// - Cette fonction ne doit pas être appelée concurremment avec d'autres opérations sur l'allocateur
117    unsafe fn allocate_more_slabs(&self, slab_index: usize) -> Result<(), ()> {
118
119        // Récupération de la référenc emutable 
120        let inner = &mut *self.inner.get();
121        // récupération de la taille des blocs
122        let slab_size = SLAB_SIZES[slab_index];
123        
124        // Calculer combien de slabs on peut créer
125        let slabs_to_create = 10; // Créons 10 slabs à la fois
126        let required_memory = slab_size * slabs_to_create;
127        
128        // Trouver une zone libre dans notre mémoire
129        let current_alloc = inner.memory_start + inner.allocated_counts.iter().enumerate()
130            .map(|(i, &count)| count * SLAB_SIZES[i])
131            .sum::<usize>();
132        
133        // Vérifie si il reste asser de place
134        if current_alloc + required_memory > inner.memory_end {
135            return Err(());  // Plus assez de mémoire
136        }
137        
138        // Créer les nouveaux slabs et les ajouter à la liste libre
139        for i in 0..slabs_to_create {
140            // Assurer que l'adresse du bloc est correctement alignée
141            let block_addr = current_alloc + i * slab_size;
142            // les allignement mémoires
143            let alignment = mem::align_of::<SlabBlock>();
144            let aligned_addr = (block_addr + alignment - 1) & !(alignment - 1);
145            // Transformation de l'adresse aligné en "pointeur vers un bloc mémoire"
146            let block_ptr = aligned_addr as *mut SlabBlock;
147            
148            // Ajouter à la liste libre  
149            (*block_ptr).next = inner.free_lists[slab_index];
150            inner.free_lists[slab_index] = block_ptr;
151        }
152        // Mise à jour du compteur
153        inner.allocated_counts[slab_index] += slabs_to_create;
154        Ok(())
155    }
156}

158/// Implémentation thread-safe de GlobalAlloc
159/// # Safety
160/// On dit ici que SlabAllocator implémente le trait GlobalAlloc, ce qui permet de l’utiliser comme allocateur global pour tout le programme.
161/// ⚠️ unsafe car on manipule la mémoire manuellement via des pointeurs
162unsafe impl GlobalAlloc for SlabAllocator {
163    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
164        // Utiliser UnsafeCell.get() pour accéder de manière sûre aux données mutables
165        let inner = &mut *self.inner.get();
166        
167        // Trouver la taille de slab appropriée
168        if let Some(slab_index) = self.find_slab_index(&layout) {
169            // Vérifier s'il y a un bloc libre disponible
170            if inner.free_lists[slab_index].is_null() {
171                // Allouer plus de slabs si nécessaire
172                if self.allocate_more_slabs(slab_index).is_err() {
173                    return ptr::null_mut();
174                }
175            }
176            
177            // Prendre un bloc de la liste libre
178            let block = inner.free_lists[slab_index];
179            inner.free_lists[slab_index] = (*block).next;
180            
181            // Vérifier l'alignement
182            let ptr = block as *mut u8;
183            if (ptr as usize) % layout.align() != 0 {
184                // Le pointeur n'est pas aligné correctement, on le désalloue
185                // et on retourne une erreur
186                (*block).next = inner.free_lists[slab_index];
187                inner.free_lists[slab_index] = block;
188                return ptr::null_mut();
189            }
190            // Sinon, on retourne le pointeur vers la mémoire prête à être utilisé
191            return ptr;
192        }
193        
194        // Aucune taille de slab appropriée
195        ptr::null_mut()
196    }
197    /// Fonction appelée quand on veux désallouer uen zone mémoire
198    unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout) {
199        // Utiliser UnsafeCell.get() pour accéder de manière sûre aux données mutables
200        let inner = &mut *self.inner.get();
201        
202        // Trouver la taille de slab appropriée
203        if let Some(slab_index) = self.find_slab_index(&layout) {
204            // Ajouter le bloc à la liste libre
205            let block = ptr as *mut SlabBlock;
206            // ajout du bloc en tête de liste
207            (*block).next = inner.free_lists[slab_index];
208            inner.free_lists[slab_index] = block;
209        }
210    }
211}

213// Définir un allocateur global si la feature 'alloc' est activée
214#[cfg(feature = "alloc")]
215pub struct GlobalSlabAllocator {
216    // Encepsulation de l'allocateru standard en Unsafecell pour le rendre mutable
217    inner: UnsafeCell<SlabAllocator>,
218    // un booléan(truc qui dit oui ou non) qui vérrifie si l'allocateur a déja été initialisé
219    initialized: core::sync::atomic::AtomicBool,
220}
221
222/// une structure est thread-safe car `UnsafeCell` n'est pas Sync par défaut
223#[cfg(feature = "alloc")]
224unsafe impl Sync for GlobalSlabAllocator {}
225
226#[cfg(feature = "alloc")]
227impl GlobalSlabAllocator {
228    /// Fonction constante pour initialisé une instance vide à la compilation
229    pub const fn new() -> Self {
230        GlobalSlabAllocator {
231            /// Valeurinitiale = alllocateur vide ( adresse 0, taill e 0)
232            inner: UnsafeCell::new(SlabAllocator::new(0, 0)),
233            initialized: core::sync::atomic::AtomicBool::new(false),
234        }
235    }
236    
237    /// initialisation de l'allocateur avec une plage mémoire donnée
238    pub fn init(&self, start: usize, size: usize) {
239        use core::sync::atomic::Ordering;
240        
241        /// si pas encore initialisé
242        if !self.initialized.load(Ordering::Acquire) {
243            unsafe {
244                // récupère l'accès mutable à l'allocateur
245                let allocator = &mut *self.inner.get();
246                *allocator = SlabAllocator::new(start, size); // crée un nouvelle allocateur
247                allocator.init(); // initialise les structures internes
248                self.initialized.store(true, Ordering::Release); // signale que c'est pret
249            }
250        }
251    }
252}

254#[cfg(feature = "alloc")]
255unsafe impl GlobalAlloc for GlobalSlabAllocator {
256    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
257        use core::sync::atomic::Ordering;
258        // Si l'allocateur n'est pas initialisé , on fait rien
259        if !self.initialized.load(Ordering::Acquire) {
260            return ptr::null_mut();
261        }
262        // laisse l'allocation à l'allocateur interne
263        (*self.inner.get()).alloc(layout)
264    }
265    
266    unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout) {
267        use core::sync::atomic::Ordering;
268        
269        if self.initialized.load(Ordering::Acquire) {
270            (*self.inner.get()).dealloc(ptr, layout);
271        }
272    }
273}
